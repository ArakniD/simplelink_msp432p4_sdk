32 /*!*****************************************************************************
33  * @file Timer.h
34  * @brief Timer driver
35  *
36  * @anchor ti_drivers_Timer_Overview
37  * # Overview
38  * The timer driver allows you to measure elapsed time with simple and
39  * portable APIs.This driver does not have PWM or capture functionalities.
40  * These functionalities are addressed in both the capture and PWM driver.
41  *
42  * The timer driver also handles the general purpose timer resource allocation.
43  * For each driver that requires use of a general purpose timer, it calls
44  * Timer_open() to occupy the specified timer, and calls Timer_close() to
45  * release the occupied timer resource.
46  *
47  * @anchor ti_drivers_Timer_Usage
48  * # Usage
49  *
50  * This documentation provides a basic @ref ti_drivers_Timer_Synopsis
51  * "usage summary" and a set of @ref ti_drivers_Timer_Examples "examples"
52  * in the form of commented code fragments. Detailed descriptions of the
53  * APIs are provided in subsequent sections.
54  *
55  * @anchor ti_drivers_Timer_Synopsis
56  * ## Synopsis
57  * @anchor ti_drivers_Timer_Synopsis_Code
58  * @code
59  * // Import Timer Driver definitions
60  * #include <ti/drivers/Timer.h>
61  *
62  * Timer_Handle handle;
63  * Timer_Params params;
64  *
65  * Timer_Params_init(&params);
66  * params.periodUnits = Timer_PERIOD_HZ;
67  * params.period = 1000;
68  * params.timerMode = Timer_CONTINUOUS_CALLBACK;
69  * params.timerCallback = UserCallbackFunction;
70  *
71  * handle = Timer_open(Board_TIMER0, &params);
72 
73 
74 
75  *
76  * @code
77  * // Import Timer Driver definitions
78  * #include <ti/drivers/Timer.h>
79  *
80  * Timer_Handle handle;
81  * Timer_Params params;
82  *
83  * // Initialize Timer parameters
84  * Timer_Params_init(&params);
85  * params.periodUnits = Timer_PERIOD_HZ;
86  * params.period = 1000;
87  * params.timerMode = Timer_CONTINUOUS_CALLBACK;
88  * params.timerCallback = UserCallbackFunction;
89  *
90  * // Open Timer instance
91  * handle = Timer_open(Board_TIMER0, &params);
92  *
93  * sleep(10000);
94  *
95  * Timer_stop(handle);
96  * @endcode
97  *
98  * <hr>
99  * @anchor ti_drivers_Timer_Examples
100  * # Examples
101  *
102  * @li @ref ti_drivers_Timer_Examples_open "Opening a Timer Instance"
103  * @li @ref ti_drivers_Timer_Examples_mode "Configuring Timer mode and period"
104  *
105  * @anchor ti_drivers_Timer_Examples_open
106  * ## Opening a Timer instance
107  *
108  * @code
109  * Timer_Handle handle;
110  * Timer_Params params;
111  *
112  * Timer_Params_init(&params);
113  * handle = Timer_open(Board_TIMER0, &params);
114  *
115  * if (handle == NULL) {
116  * // Timer_open() failed
117  * while (1);
118  * }
119  @endcode
120  *
121  * @anchor ti_drivers_Timer_Examples_mode
122  * ##Configuring Timer mode and period
123  *
124  * The following example code opens a timer in continuous callback mode. The
125  * period is set to 1000 Hz.
126  *
127  * @code
128  * Timer_Handle handle;
129  * Timer_Params params;
130  *
131  * Timer_Params_init(&params);
132  * params.periodUnits = Timer_PERIOD_HZ;
133  * params.period = 1000;
134  * params.timerMode = Timer_CONTINUOUS_CALLBACK;
135  * params.timerCallback = UserCallbackFunction;
136  *
137  * handle = Timer_open(Board_TIMER0, &params);
138  *
139  * if (handle == NULL) {
140  * // Timer_open() failed
141  * while (1);
142  * }
143  *
144  * status = Timer_start(handle);
145  *
146  * if (status == Timer_STATUS_ERROR) {
147  * //Timer_start() failed
148  * while (1);
149  * }
150  *
151  * sleep(10000);
152  *
153  * Timer_stop(handle);
154  * @endcode
155  *
156  * ### Initializing the Timer Driver #
157  *
158  * Timer_init() must be called before any other timer APIs. This function
159  * calls the device implementation's timer initialization function, for each
160  * element of Timer_config[].
161  *
162  * <hr>
163  * @anchor ti_drivers_Timer_Configuration
164  * # Configuration
165  *
166  * Refer to the @ref driver_configuration "Driver's Configuration" section
167  * for driver configuration information.
168  * <hr>
169 *******************************************************************************
