    40    /*!
    41     *  ======== SecondsClock ========
    42     *  Seconds module that uses Clock as its' time base
    43     *
    44     */
    45    
    46    @ModuleStartup
    47    
    48    module SecondsClock inherits ti.sysbios.interfaces.ISeconds
    49    {
    50    
    51    internal:
    52        /*!
    53         *  @_nodoc
    54         *  ======== increment ========
    55         *  bump time every second
    56         */
    57        Void increment(UArg arg);
    58    
    59        struct Module_State {
    60            UInt32        seconds;
    61            Clock.Object  clock;
    62            UInt32        c1;
    63            UInt32        c2;
    64            Int32         c1Inc;
    65            UInt32        count1;
    66            UInt32        count2;
    67            UInt32        ticks;
    68        }
    69    }
