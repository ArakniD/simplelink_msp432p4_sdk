    36    package ti.sysbios.gates
    37    
    38    import xdc.rov.ViewInfo;
    39    import xdc.runtime.Assert;
    40    
    41    /*!
    42     *  ======== GateSwi ========
    43     *  Software Interrupt Gate.
    44     *
    45     *  GateSwi uses disabling and enabling of software interrupts as the
    46     *  resource locking  mechanism. This gate can be used whenever the resource
    47     *  is being shared by Swis or Tasks. This gate cannot be used by a Hwi.
    48     *
    49     *  The duration between the enter and leave should be as short as possible
    50     *  to minimize Swi latency.
    51     *
    52     *  GateSwi inherits from xdc.runtime.IGateProvider, therefore supports nesting.
    53     *
    54     *  @p(html)
    55     *  <h3> Calling Context </h3>
    56     *  <table border="1" cellpadding="3">
    57     *    <colgroup span="1"></colgroup> <colgroup span="5" align="center">
    58     *    </colgroup>
    59     *
    60     *    <tr><th> Function </th><th>  Hwi   </th><th>  Swi   </th><th>  Task  </th>
    61     *    <th>  Main  </th><th>  Startup  </th></tr>
    62     *    <!--                                           -->
    63     *    <tr><td> {@link #Params_init} </td><td>   Y    </td><td>   Y    </td>
    64     *    <td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
    65     *    <tr><td> {@link #query}       </td><td>   Y    </td><td>   Y    </td>
    66     *    <td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
    67     *    <tr><td> {@link #construct}   </td><td>   Y    </td><td>   Y    </td>
    68     *    <td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
    69     *    <tr><td> {@link #create}      </td><td>   N*   </td><td>   N*   </td>
    70     *    <td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
    71     *    <tr><td> {@link #delete}      </td><td>   N*   </td><td>   N*   </td>
    72     *    <td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
    73     *    <tr><td> {@link #destruct}    </td><td>   Y    </td><td>   Y    </td>
    74     *    <td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
    75     *    <tr><td> {@link #enter}       </td><td>   N    </td><td>   Y    </td>
    76     *    <td>   Y    </td><td>   Y**  </td><td>   N    </td></tr>
    77     *    <tr><td> {@link #leave}       </td><td>   N    </td><td>   Y    </td>
    78     *    <td>   Y    </td><td>   Y**  </td><td>   N    </td></tr>
    79     *    <tr><td colspan="6"> Definitions: <br />
    80     *       <ul>
    81     *         <li> <b>Hwi</b>: API is callable from a Hwi thread. </li>
    82     *         <li> <b>Swi</b>: API is callable from a Swi thread. </li>
    83     *         <li> <b>Task</b>: API is callable from a Task thread. </li>
    84     *         <li> <b>Main</b>: API is callable during any of these phases: </li>
    85     *           <ul>
    86     *             <li> In your module startup after this module is started
    87     *    (e.g. GateSwi_Module_startupDone() returns TRUE). </li>
    88     *             <li> During xdc.runtime.Startup.lastFxns. </li>
    89     *             <li> During main().</li>
    90     *             <li> During BIOS.startupFxns.</li>
    91     *           </ul>
    92     *         <li> <b>Startup</b>: API is callable during any of these phases:</li>
    93     *           <ul>
    94     *             <li> During xdc.runtime.Startup.firstFxns.</li>
    95     *             <li> In your module startup before this module is started
    96     *    (e.g. GateSwi_Module_startupDone() returns FALSE).</li>
    97     *           </ul>
    98     *       <li> <b>*</b>:  Assuming blocking Heap is used for creation. </li>
    99     *       <li> <b>**</b>: Must be used in enter/leave pairs. </li>
   100     *       </ul>
   101     *    </td></tr>
   102     *
   103     *  </table>
   104     *  @p
   105     */
   106    
   107    module GateSwi inherits xdc.runtime.IGateProvider
   108    {
   109    
   110        /*!
   111         *  ======== BasicView ========
   112         *  @_nodoc
   113         */
   114        metaonly struct BasicView {
   115            String label;
   116        }
   117    
   118        /*!
   119         *  ======== rovViewInfo ========
   120         *  @_nodoc
   121         */
   122        @Facet
   123        metaonly config ViewInfo.Instance rovViewInfo =
   124            ViewInfo.create({
   125                viewMap: [
   126                [
   127                    'Basic',
   128                    {
   129                        type: ViewInfo.INSTANCE,
   130                        viewInitFxn: 'viewInitBasic',
   131                        structName: 'BasicView'
   132                    }
   133                ],
   134                ]
   135            });
   136    
   137        /*!
   138         * Asserted when {@link #enter} or {@link #leave} is called from a
   139         * hardware interrupt thread.
   140         */
   141        config Assert.Id A_badContext = {
   142                msg: "A_badContext: bad calling context. May not be entered from a hardware interrupt thread."
   143        };
   144    
   145    instance:
   146    
   147        /*!
   148         *  ======== enter ========
   149         *  Disable Swi Scheduling
   150         *
   151         *  GateSwi_enter() disables all Swi functions from running until
   152         *  GateSwi_leave() is called. Hardware interrupts can still run.
   153         *
   154         *  Read the discussion of the side effects of disabling the Swi
   155         *  scheduler {@link ti.sysbios.knl.Swi#disable here}.
   156         */
   157        override IArg enter();
   158    
   159        /*!
   160         *  ======== leave ========
   161         *  Re-enable Swi Scheduling
   162         *
   163         *  GateSwi_enter() disables all Swi functions from running until
   164         *  GateSwi_leave() is called. Hardware interrupts can still run.
   165         *
   166         *  Read the discussion of the side effects of disabling the Swi
   167         *  scheduler {@link ti.sysbios.knl.Swi#disable here}.
   168         */
   169        override Void leave(IArg key);
   170    
   171    internal:
   172    
   173        /*!
   174         *  @_nodoc
   175         *  ======== Instance_State ========
   176         *  XDC requires this be defined, even though it's empty.
   177         */
   178        struct Instance_State {
   179        };
   180    }
