    36    package ti.sysbios.posix;
    37    
    38    /*!
    39     *  ======== Settings ========
    40     *  Obsolete module for POSIX support
    41     *
    42     *  Please refer to ti.posix.rtos.Settings for POSIX support.
    43     */
    44    
    45    metaonly module Settings
    46    {
    47        /*!
    48         *  ======== supportsMutexPriority ========
    49         *  Include support for pthread mutex priority protocols. Default is false.
    50         *
    51         *  Set to true to enable pthread mutex priority protocols.
    52         *  If set to true, PTHREAD_PRIO_PROTECT and PTHREAD_PRIO_INHERIT mutexes
    53         *  will be available.  The default value is set to false to reduce
    54         *  code size.
    55         */
    56        config Bool supportsMutexPriority = false;
    57    
    58        /*!
    59         *  ======== debug ========
    60         *  Enable debug functions. Default is true.
    61         *  @_nodoc
    62         */
    63        config Bool debug = false;
    64    }
