    38    package ti.sysbios.family.arm.m3;
    39    
    40    /*!
    41     *  ======== Power ========
    42     *  Power support for m3.
    43     */
    44    
    45    @Template("./Power.xdt")
    46    
    47    module Power inherits ti.sysbios.interfaces.IPower
    48    {
    49        /*!
    50         *  ======== idle ========
    51         *  Idle CPU when threads blocked waiting for an interrupt?
    52         */
    53        override config Bool idle = false;
    54    
    55        /*!
    56         *  ======== idleCPU ========
    57         *  Function to automatically idle the CPU during idle time
    58         *
    59         *  When the `{@link #idle}` configuration parameter is set, this function
    60         *  will be added to the list of Idle loop functions.  When called from the
    61         *  `Idle` loop, it will transition the CPU into the idle mode specified
    62         *  by the `{@link #idleMode}` configuration parameter.  The CPU will stay
    63         *  in the corresponding low power mode until the next interrupt occurs.
    64         */
    65        @DirectCall
    66        Void idleCPU();
    67    }
