    40    /*!
    41     *  ======== Exception ========
    42     *  (Deprecated!) Exception Module
    43     *
    44     *  Do not use this module!
    45     *
    46     *  The functionality of the Exception module has been fully integrated
    47     *  into the {@link Hwi} module.
    48     *
    49     *  The Exception module is a basic Cortex M3 exception handler.
    50     *  No attempt is made to return from an exception as these are
    51     *  generally fatal in nature.
    52     *
    53     *  When used, the Exception module will override the default 
    54     *  exception handlers provided by the {@link Hwi} module.
    55     *  This can be done by adding the following to your configuration
    56     *  script:
    57     *
    58     *  @p(code)
    59     *  Exception = xdc.useModule('ti.sysbios.family.m3.Exception');
    60     *  @p
    61     */
    62    
    63    metaonly module Exception
    64    {
    65    
    66        /*!
    67         *  If true, use System_printf() to dump exception context.
    68         */
    69        config Bool doPrint = false;
    70    }
