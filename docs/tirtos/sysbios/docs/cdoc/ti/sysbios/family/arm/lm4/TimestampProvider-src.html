    38    package ti.sysbios.family.arm.lm4;
    39    
    40    /*!
    41     *  ======== TimestampProvider ========
    42     *  Stellaris LM4 TimestampProvider
    43     *
    44     *  Stellaris LM4 TimestampProvider delegate for use
    45     *  with {@link xdc.runtime.Timestamp}.
    46     *
    47     *  Implements {@link ti.sysbios.interfaces.ITimestampProvider}
    48     *
    49     *  A timer managed by the {@link Timer Stellaris LM4 Timer} module
    50     *  is used as the timestamp source.
    51     *
    52     *  @p(html)
    53     *  <h3> Calling Context </h3>
    54     *  <table border="1" cellpadding="3">
    55     *    <colgroup span="1"></colgroup> <colgroup span="5" align="center"></colgroup>
    56     *
    57     *    <tr><th> Function                 </th><th>  Hwi   </th><th>  Swi   </th><th>  Task  </th><th>  Main**  </th><th>  Startup***  </th></tr>
    58     *    <!--                                                                                                                 -->
    59     *    <tr><td> {@link #get32}           </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
    60     *    <tr><td> {@link #get64}           </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
    61     *    <tr><td> {@link #getFreq}         </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
    62     *    <tr><td colspan="6"> Definitions: <br />
    63     *       <ul>
    64     *         <li> <b>Hwi</b>: API is callable from a Hwi thread. </li>
    65     *         <li> <b>Swi</b>: API is callable from a Swi thread. </li>
    66     *         <li> <b>Task</b>: API is callable from a Task thread. </li>
    67     *         <li> <b>Main</b>: API is callable during any of these phases: </li>
    68     *           <ul>
    69     *             <li> In your module startup after this module is started (e.g. TimestampProvider_Module_startupDone() returns TRUE). </li>
    70     *             <li> During xdc.runtime.Startup.lastFxns. </li>
    71     *             <li> During main().</li>
    72     *             <li> During BIOS.startupFxns.</li>
    73     *           </ul>
    74     *         <li> <b>Startup</b>: API is callable during any of these phases:</li>
    75     *           <ul>
    76     *             <li> During xdc.runtime.Startup.firstFxns.</li>
    77     *             <li> In your module startup before this module is started (e.g. TimestampProvider_Module_startupDone() returns FALSE).</li>
    78     *           </ul>
    79     *       </ul>
    80     *    </td></tr>
    81     *  </table>
    82     *  @p
    83     */
    84    
    85    @ModuleStartup
    86    
    87    module TimestampProvider inherits ti.sysbios.interfaces.ITimestamp
    88    {
    89        /*!
    90         *  Set this parameter when you want TimestampProvider to use
    91         *  the same timer as the Clock module.
    92         *
    93         *  This parameter overrides the {@link #timerId} setting.
    94         */
    95        config Bool useClockTimer = false;
    96    
    97        /*!
    98         *  TimestampProvider timer id.
    99         *
   100         *  When not using the Clock timer, TimestampProvider will
   101         *  create an {@link Timer} using this timer id.
   102         */
   103        config UInt timerId = Timer.ANY;
   104    
   105        /*!@_nodoc
   106         *  ======== startTimer ========
   107         *  Internal use. Function that starts timer before main
   108         */
   109        Void startTimer();
   110    
   111        /*!@_nodoc
   112         *  ======== rolloverFunc ========
   113         *  Internal use. Timer rollover interrupt handler.
   114         */
   115        Void rolloverFunc(UArg unused);
   116    
   117    internal:   /* not for client use */
   118    
   119        /*!@_nodoc
   120         *  ======== initTimerHandle ========
   121         *  firstFxn used to ensure that Module->timer is initialized
   122         *  before Timestamp_get32() is invoked
   123         */
   124        Void initTimerHandle();
   125    
   126        struct Module_State {
   127            Timer.Handle    timer;          /* timer instance used */
   128            volatile UInt32 hi;             /* upper 32 bits */
   129        }
   130    }
