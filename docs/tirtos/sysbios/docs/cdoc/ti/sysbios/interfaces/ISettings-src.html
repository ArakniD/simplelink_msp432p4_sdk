    36    package ti.sysbios.interfaces;
    37    
    38    /*!
    39     *  ======== ISettings ========
    40     *  Interface for family-specific Settings modules.
    41     *
    42     *  Every new device family must implement this interface.
    43     */
    44    metaonly interface ISettings
    45    {
    46        /*! returns the family-specific Core module delegate */
    47        String getDefaultCoreDelegate();
    48        
    49        /*! returns the family-specific Hwi module delegate */
    50        String getDefaultHwiDelegate();
    51    
    52        /*! returns the family-specific Timer module delegate */
    53        String getDefaultTimerDelegate();
    54        
    55        /*! returns the family-specific Clock Timer module delegate */
    56        String getDefaultClockTimerDelegate();
    57    
    58        /*! returns the family-specific Timer Support module delegate */
    59        String getDefaultTimerSupportDelegate();
    60    
    61        /*! returns the family-specific Timestamp module delegate */
    62        String getDefaultTimestampDelegate();
    63    
    64        /*! returns the family-specific TaskSupport module delegate */
    65        String getDefaultTaskSupportDelegate();
    66    
    67        /*! returns the family-specific IntrinsicsSupport module delegate */
    68        String getDefaultIntrinsicsSupportDelegate();
    69    
    70        /*! returns the family-specific CacheSupport module delegate */
    71        String getDefaultCacheDelegate();
    72    
    73        /*! returns the family-specific Power module delegate */
    74        String getDefaultPowerDelegate();
    75    
    76        /*! returns the family-specific Seconds module delegate */
    77        String getDefaultSecondsDelegate();
    78    
    79        /*! returns the family-specific Boot module. */
    80        String getDefaultBootModule();
    81    
    82        /*! returns the family-specific Mmu module. */
    83        String getDefaultMmuModule();
    84        
    85        /*! Returns the path to the family-specific Grace page. */
    86        String getFamilySettingsXml();
    87        
    88        /*! returns the family-specific Clock tickPeriod */
    89        UInt32 getDefaultClockTickPeriod();
    90    }
