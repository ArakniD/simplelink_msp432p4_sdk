    37    /*!
    38     *  ======== ti.sysbios.interfaces ========
    39     *  Contains all interface specifications.
    40     *
    41     *  For the proxy-delegate mechanism between generic modules and the 
    42     *  modules in {@link ti.sysbios.family} to work properly, both the 
    43     *  proxy and the delegate modules must be implementations of a common 
    44     *  interface specification. The Timer, Hwi, Cache and Seconds interface
    45     *  specifications reside in ti.sysbios.interfaces and are 
    46     *  {@link ti.sysbios.interfaces.ITimer}, 
    47     *  {@link ti.sysbios.interfaces.IHwi},
    48     *  {@link ti.sysbios.interfaces.ICache}, and
    49     *  {@link ti.sysbios.interfaces.ISeconds} respectively.
    50     *
    51     *  These interface specifications define a minimum set of general APIs
    52     *  that, it is  believed, will satisfy a vast majority of application
    53     *  requirements. 
    54     *  
    55     *  The I<mod>Support interfaces define "internal" interfaces used by the
    56     *  <mod> modules in their implementation.  These interfaces simplify the
    57     *  process of porting to a new architecture; one only needs to implement
    58     *  the I<mod>Support interface to enable the full functionality of the
    59     *  ti.sysbios.<mod> module.
    60     *
    61     */
    62    package ti.sysbios.interfaces [2,0,0,0] {
    63        interface ICore;
    64        interface IHwi;
    65        interface ITaskSupport;
    66        interface ITimer;
    67        interface ITimerSupport;
    68        interface ITimestamp;
    69        interface IIntrinsicsSupport;
    70        interface ICache;
    71        interface ISettings;
    72        interface IPower;
    73        interface IRomDevice;
    74        interface ISeconds;
    75    }
