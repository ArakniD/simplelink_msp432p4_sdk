    44    /*! ======== BIOS ========
    45     *  SYS/BIOS Top-Level Manager
    46     *
    47     *  This module is responsible for setting up global parameters
    48     *  pertaining to SYS/BIOS and for performing the SYS/BIOS startup
    49     *  sequence.
    50     *
    51     *  SYS/BIOS configures the
    52     *  {@link xdc.runtime.Memory#defaultHeapInstance Memory.defaultHeapInstance}
    53     *  using a {@link ti.sysbios.heaps.HeapMem HeapMem} instance of size
    54     *  {@link #heapSize}.
    55     *
    56     *  The SYS/BIOS startup sequence is logically divided into two phases: those
    57     *  operations that occur prior to the application's "main()" function being
    58     *  called, and those operations that are performed after the application's
    59     *  "main()" function is invoked.
    60     *
    61     *  The "before main()" startup sequence is governed completely by the RTSC
    62     *  runtime package's {@link xdc.runtime.Startup Startup} module.
    63     *
    64     *  The "after main()" startup sequence is governed by SYS/BIOS and is
    65     *  initiated by an explicit call to the {@link #start BIOS_start()} function
    66     *  at the end of the application's main() function.
    67     *
    68     *  Control points are provided at various places in each of the two startup
    69     *  sequences for user startup operations to be inserted.
    70     *
    71     *  The RTSC runtime startup sequence is as follows:
    72     *
    73     *  @p(nlist)
    74     *  - Immediately after CPU reset, perform target-specific CPU
    75     *  initialization (beginning at c_int00).
    76     *  - Prior to cinit(), run the user-supplied "reset functions"
    77     *  (see {@link xdc.runtime.Reset#fxns Reset.fxns}).
    78     *  - Run cinit() to initialize C runtime environment.
    79     *  - Run the user-supplied "first functions"
    80     *  (see {@link xdc.runtime.Startup#firstFxns Startup.firstFxns}).
    81     *  - Run all the module initialization functions.
    82     *  - Run pinit().
    83     *  - Run the user-supplied "last functions"
    84     *  (see {@link xdc.runtime.Startup#lastFxns Startup.lastFxns}).
    85     *  - Run main().
    86     *  @p
    87     *
    88     *  The SYS/BIOS startup sequence begins at the end of main() when
    89     *  BIOS_start() is called:
    90     *
    91     *  @p(nlist)
    92     *  - Run the user-supplied "startup functions"
    93     *  (see {@link #startupFxns BIOS.startupFxns}).
    94     *  - Enable Hardware Interrupts.
    95     *  - Enable Software Interrupts. If the system supports Software Interrupts
    96     *  (Swis) (see {@link #swiEnabled BIOS.swiEnabled}), then the SYS/BIOS
    97     *  startup sequence enables Swis at this point.
    98     *  - Timer Startup. If the system supports Timers, then at this point all
    99     *  statically configured timers are initialized per their
   100     *  user-configuration.
   101     *  If a timer was configured to start "automatically", it is started here.
   102     *  - Task Startup. If the system supports Tasks
   103     *  (see {@link #taskEnabled BIOS.taskEnabled}),
   104     *  then task scheduling begins here. If there are no statically or
   105     *  dynamically created Tasks in the system, then execution proceeds
   106     *  directly to the Idle loop.
   107     *  @p
   108     *
   109     *  @a(Note)
   110     *  Local variables defined in main() no longer exist once BIOS_start() is
   111     *  called. The RAM where main's local variables reside is reassigned for
   112     *  use as the interrupt stack during the execution of BIOS_start().
   113     *
   114     *  Below is a configuration script excerpt that installs a user-supplied
   115     *  startup function at every possible control point in the RTSC and
   116     *  SYS/BIOS startup
   117     *  sequence:
   118     *
   119     *  @p(code)
   120     *  // get handle to xdc Startup module
   121     *  var Startup = xdc.useModule('xdc.runtime.Startup');
   122     *
   123     *  // install "reset function"
   124     *  Startup.resetFxn = '&myReset';
   125     *
   126     *  // install a "first function"
   127     *  var len = Startup.firstFxns.length
   128     *  Startup.firstFxns.length++;
   129     *  Startup.firstFxns[len] = '&myFirst';
   130     *
   131     *  // install a "last function"
   132     *  var len = Startup.lastFxns.length
   133     *  Startup.lastFxns.length++;
   134     *  Startup.lastFxns[len] = '&myLast';
   135     *
   136     *  // get handle to SYS/BIOS module
   137     *  var BIOS = xdc.useModule('ti.sysbios.BIOS');
   138     *
   139     *  // install a SYS/BIOS startup function
   140     *  BIOS.addUserStartupFunction('&myBiosStartup');
   141     *  @p
   142     *
   143     *  @p(html)
   144     *  <h3> Calling Context </h3>
   145     *  <table border="1" cellpadding="3">
   146     *    <colgroup span="1"></colgroup> <colgroup span="5" align="center">
   147     *    </colgroup>
   148     *
   149     *    <tr><th> Function                 </th><th>  Hwi   </th><th>  Swi   </th>
   150     *    <th>  Task  </th><th>  Main  </th><th>  Startup  </th></tr>
   151     *    <!--                                        -->
   152     *    <tr><td> {@link #getCpuFreq}      </td><td>   Y    </td><td>   Y    </td>
   153     *    <td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
   154     *    <tr><td> {@link #getThreadType}   </td><td>   Y    </td><td>   Y    </td>
   155     *    <td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   156     *    <tr><td> {@link #setCpuFreq}      </td><td>   Y    </td><td>   Y    </td>
   157     *    <td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
   158     *    <tr><td> {@link #start}      </td><td>   N    </td><td>   N    </td>
   159     *    <td>   N    </td><td>   Y    </td><td>   N    </td></tr>
   160     *    <tr><td colspan="6"> Definitions: <br />
   161     *       <ul>
   162     *         <li> <b>Hwi</b>: API is callable from a Hwi thread. </li>
   163     *         <li> <b>Swi</b>: API is callable from a Swi thread. </li>
   164     *         <li> <b>Task</b>: API is callable from a Task thread. </li>
   165     *         <li> <b>Main</b>: API is callable during any of these phases: </li>
   166     *           <ul>
   167     *             <li> In your module startup after this module is started
   168     *                  (e.g. BIOS_Module_startupDone() returns TRUE). </li>
   169     *             <li> During xdc.runtime.Startup.lastFxns. </li>
   170     *             <li> During main().</li>
   171     *             <li> During BIOS.startupFxns.</li>
   172     *           </ul>
   173     *         <li> <b>Startup</b>: API is callable during any of these phases:</li>
   174     *           <ul>
   175     *             <li> During xdc.runtime.Startup.firstFxns.</li>
   176     *             <li> In your module startup before this module is started
   177     *                  (e.g. BIOS_Module_startupDone() returns FALSE).</li>
   178     *           </ul>
   179     *       </ul>
   180     *    </td></tr>
   181     *
   182     *  </table>
   183     *  @p
   184     */
   185

   193         *  ======== ThreadType ========
   194         *  Current thread type definitions
   195         *
   196         *  These values are returned by {@link #getThreadType BIOS_getThreadType}.
   197         *
   198         *  @see #getThreadType
   199         */
   200        enum ThreadType {
   201            ThreadType_Hwi,         /*! Current thread is a Hwi */
   202            ThreadType_Swi,         /*! Current thread is a Swi */
   203            ThreadType_Task,        /*! Current thread is a Task */
   204            ThreadType_Main         /*! Current thread is Boot/Main */
   205        };
   206    
   207        /*!
   208         *  ======== RtsLockType ========
   209         *  Type of Gate to use in the TI RTS library
   210         *
   211         *  @field(NoLocking) no gate is added to the RTS library.  In this case,
   212         *  the application needs to be careful to always serialize access to the
   213         *  inherently  non-reentrant ANSI C functions (such as `malloc()`,
   214         *  `printf()`, etc.).
   215         *
   216         *  @field(GateHwi) Interrupts are disabled and restored to maintain
   217         *  re-entrancy.  This is a very efficient lock but will also result in
   218         *  unbounded interrupt latency times.  If real-time response to interrupts
   219         *  is important, you should not use this gate to lock the RTS library.
   220         *
   221         *  @field(GateSwi) Swis are disabled and restored to maintain
   222         *  re-entrancy.
   223         *
   224         *  @field(GateMutex) A single mutex is used to maintain re-entrancy.
   225         *
   226         *  @field(GateMutexPri) A single priority inheriting mutex is used to
   227         *  maintain re-entrancy.
   228         *
   229         *  @see #rtsGateType
   230         */
   231        enum RtsLockType {
   232            NoLocking,
   233            GateHwi,
   234            GateSwi,
   235            GateMutex,
   236            GateMutexPri
   237        };

   239        /*!
   240         *  ======== LibType ========
   241         *  SYS/BIOS library selection options
   242         *
   243         *  This enumeration defines all the SYS/BIOS library types
   244         *  supported by the product.  You can select the library type by setting
   245         *  the {@link #libType BIOS.libType} configuration parameter.
   246         *
   247         *  @field(LibType_Instrumented) The library is built with logging and
   248         *  assertions enabled.
   249         *
   250         *  @field(LibType_NonInstrumented) The library is built with logging and
   251         *  assertions disabled.
   252         *
   253         *  @field(LibType_Custom) The library is built using the options
   254         *  specified by {@link #customCCOpts}. Program optimization is performed
   255         *  to reduce the size of the executable and improve its performance.
   256         *  Enough debug information is retained to allow you to step through the
   257         *  application code in CCS and locate global variables.
   258         *
   259         *  @field(LibType_Debug) This setting is similar to the LibType_Custom
   260         *  setting, however, no program optimization is performed. The resulting
   261         *  executable is fully debuggable, and you can step into SYS/BIOS code.
   262         *  The tradeoff is that the executable is larger and runs slower than
   263         *  builds that use the LibType_Custom option.
   264         *
   265         *  @see #libType
   266         */
   267        enum LibType {
   268            LibType_Instrumented,           /*! Instrumented (Asserts and Logs enabled) */
   269            LibType_NonInstrumented,        /*! Non-instrumented (Asserts and Logs disabled) */
   270            LibType_Custom,                 /*! Custom (Fully configurable) */
   271            LibType_Debug                   /*! Debug (Fully configurable) */
   272        };

   352        /*!
   353         *  ======== customCCOpts ========
   354         *  Compiler options used when building a custom SYS/BIOS library
   355         *
   356         *  When {@link #libType BIOS.libType} is set to
   357         *  {@link #LibType_Custom BIOS_LibType_Custom} or
   358         *  {@link #LibType_Debug BIOS_LibType_Debug},
   359         *  this string contains the options passed to the compiler during any
   360         *  build of the SYS/BIOS sources.
   361         *
   362         *  In addition to the options
   363         *  specified by `BIOS.customCCOpts`, several `-D` and `-I` options are also
   364         *  passed to the compiler.  The options specified by `BIOS.customCCOpts` are,
   365         *  however, the first options passed to the compiler on the command line.
   366         *
   367         *  To view the custom compiler options, add the following line to your
   368         *  config script:
   369         *
   370         *  @p(code)
   371         *  print(BIOS.customCCOpts);
   372         *  @p
   373         *
   374         *  When {@link #libType BIOS.libType} is set to
   375         *  {@link #LibType_Custom BIOS_LibType_Custom},
   376         *  `BIOS.customCCOpts` is initialized to settings that create a highly
   377         *  optimized SYS/BIOS library.
   378         *
   379         *  When {@link #libType BIOS.libType} is set to
   380         *  {@link #LibType_Debug BIOS_LibType_Debug},
   381         *  `BIOS.customCCOpts` is initialized to settings that create a non-optimized
   382         *  SYS/BIOS library that can be used to single-step through the APIs with
   383         *  the CCS debugger.
   384         *
   385         *  More information about using `BIOS.customCCOpts` is provided in the
   386         *  {@link http://processors.wiki.ti.com/index.php/SYS/BIOS_FAQs SYS/BIOS FAQs}.
   387         *
   388         *  @a(Warning)
   389         *  The default value of `BIOS.customCCOpts`, which is derived from the target
   390         *  specified by your configuration, includes runtime model options
   391         *  (such as endianess) that must be the same for all sources built and
   392         *  linked into your application.  You must not change or add any options
   393         *  that can alter the runtime model specified by the default value of
   394         *  `BIOS.customCCOpts`.
   395         *
   396         *  @a(Warning)
   397         *  Setting `BIOS.libType` overwrites `BIOS.customCCOpts`. Therefore, if an
   398         *  application's *.cfg file sets both these config params, the libType must
   399         *  be set before customCCOpts so the changes to customCCOpts persist.
   400         */
   401        metaonly config String customCCOpts;

   442        /*!
   443         *  ======== cpuFreq ========
   444         *  CPU frequency in Hz
   445         *
   446         *  This configuration parameter allow SYS/BIOS to convert various
   447         *  periods between timer ticks (or instruction cycles) and real-time
   448         *  units.  For example, timer periods expressed in micro-seconds need
   449         *  to be converted into timer ticks in order to properly program the
   450         *  timers.
   451         *
   452         *  The default value of this parameter is obtained from the platform
   453         *  (the clockRate property of {@link xdc.cfg.Program#cpu Program.cpu})

   455         *
   456         *  @a(Example)
   457         *  If CPU frequency is 720MHz, the following configuration script
   458         *  configures SYS/BIOS with the proper clock frequency:
   459         *  @p(code)
   460         *     var BIOS = xdc.useModule('ti.sysbios.BIOS');
   461         *     BIOS.cpuFreq.hi = 0;
   462         *     BIOS.cpuFreq.lo = 720000000;
   463         *  @p
   464         */

   691        /*!
   692         *  ======== rtsGateType ========
   693         *  Gate to make sure TI RTS library APIs are re-entrant
   694         *
   695         *  The application gets to determine the type of gate (lock) that is used
   696         *  in the TI RTS library. The gate will be used to guarantee re-entrancy
   697         *  of the RTS APIs.
   698         *
   699         *  The type of gate depends on the type of threads that are going to
   700         *  be calling into the RTS library.  For example, if both Swi and Task
   701         *  threads are going to be calling the RTS library's printf, GateSwi
   702         *  should be used. In this case, Hwi threads are not impacted (i.e.
   703         *  disabled) during the printf calls from the Swi or Task threads.
   704         *
   705         *  If NoLocking is used, the RTS lock is not plugged and re-entrancy for
   706         *  the TI RTS library calls are not guaranteed. The application can plug
   707         *  the RTS locks directly if it wants.
   708         *
   709         *  Numerous gate types are provided by SYS/BIOS.  Each has its advantages
   710         *  and disadvantages.  The following list summarizes when each type is
   711         *  appropriate for protecting an underlying non-reentrant RTS library.
   712         *  @p(dlist)
   713         *      - {@link #GateHwi}:
   714         *        Interrupts are disabled and restored to maintain re-entrancy.
   715         *        Use if only making RTS calls from a Hwi, Swi and/or Task.
   716         *
   717         *      - {@link #GateSwi}:
   718         *        Swis are disabled and restored to maintain re-entrancy. Use if
   719         *        only making RTS calls from a Swi and/or Task.
   720         *
   721         *      - {@link #GateMutex}:
   722         *        A single mutex is used to maintain re-entrancy.  Use if only
   723         *        making RTS calls from a Task.  Blocks only Tasks that are
   724         *        also trying to execute critical regions of RTS library.
   725         *
   726         *      - {@link #GateMutexPri}:
   727         *        A priority inheriting mutex is used to maintain re-entrancy.
   728         *        Blocks only Tasks that are also trying to execute critical
   729         *        regions of RTS library.  Raises the priority of the Task that
   730         *        is executing the critical region in the RTS library to the
   731         *        level of the highest priority Task that is block by the mutex.
   732         *  @p
   733         *
   734         *  The default value of rtsGateType depends on the type of threading
   735         *  model enabled by other configuration parameters.
   736         *  If {@link #taskEnabled} is true, {@link #GateMutex} is used.
   737         *  If {@link #swiEnabled} is true and {@link #taskEnabled} is false:
   738         *  {@link #GateSwi} is used.
   739         *  If both {@link #swiEnabled} and {@link #taskEnabled} are false:
   740         *  {@link xdc.runtime#GateNull} is used.
   741         *
   742         *  If {@link #taskEnabled} is false, the user should not select
   743         *  {@link #GateMutex} (or other Task level gates). Similarly, if
   744         *  {@link #taskEnabled} and {@link #swiEnabled}are false, the user
   745         *  should not select {@link #GateSwi} or the Task level gates.
   746         */
   747        metaonly config RtsLockType rtsGateType;
