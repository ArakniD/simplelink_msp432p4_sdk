    37    import ti.uia.runtime.ServiceMgr;
    38    
    39    /*!
    40     *  ======== Settings ========
    41     *  Family setting module
    42     */
    43    metaonly module Settings
    44    {
    45        /*! returns the default transport to use on this device. */
    46        ServiceMgr.TransportType getDefaultTransport();
    47    
    48        /*! returns the family default event packet size */
    49        Int getServiceMgrEventPacketSize();
    50    
    51        /*! returns the family default control message packet size */
    52        Int getServiceMgrMsgPacketSize();
    53    
    54        /*! returns the family default transport init function */
    55        String initFxn();
    56    
    57        /*! returns the family default transport start function */
    58        String startFxn();
    59    
    60        /*! returns the family default transport recv function */
    61        String recvFxn();
    62    
    63        /*! returns the family default transport send function */
    64        String sendFxn();
    65    
    66        /*! returns the family default transport stop function */
    67        String stopFxn();
    68    
    69        /*! returns the family default transport exit function */
    70        String exitFxn();
    71    
    72        /*! returns whether the device's default transport supports control */
    73        Bool getSupportControl();
    74    }
